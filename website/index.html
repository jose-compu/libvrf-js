Quick start

npm install libvrf
import { VRF, VRFType } from 'libvrf';

const sk = VRF.create(VRFType.RSA_FDH_VRF_RSA2048_SHA256)!;
const pk = sk.getPublicKey()!;
const alpha = new TextEncoder().encode('hello');

const proof = sk.getVRFProof(alpha)!;
const [ok, beta] = pk.verifyVRFProof(alpha, proof);
console.log(ok, Buffer.from(beta).toString('hex'));
